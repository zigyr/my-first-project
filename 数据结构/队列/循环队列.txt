int equeue(Queue *q,ElemType e){
    if((q->rear+1)%MAXSIZE==q->front){
        printf("full\n");
        return 0;
    }
    q->data[q->rear]=e;
    q->rear=(q->rear+1)%MAXSIZE;
    return 1;
}

int dequeue(Queue *q,ElemType *e){
    if(q->front==q->rear){
        printf("empty\n");
        return 0;
    }
    *e=q->data[q->front];
    q->front=(q->front+1)%MAXSIZE;
    return 1;
}
